sidebar.addEventListener('click', function(e) { if (window.innerWidth > 768) return; // 桌機不干預 const a = e.target.closest('a'); if (a) { // 攔截所有連結點擊 e.preventDefault(); e.stopPropagation(); const href = a.getAttribute('href'); if (href && href.startsWith('#/')) { // 跳轉到目標頁面，但不關閉側邊欄 location.hash = href; } return; } // 處理資料夾展開/收合（原本的邏輯） const li = e.target.closest('li'); if (li) { const ul = li.querySelector('ul'); if (ul && !ul.contains(e.target)) { e.stopPropagation(); e.preventDefault(); li.classList.toggle('collapse'); } } }, true);
